2.2 Computer Vision (Webcam + ArUco → Homography)

scan รูปจากกระดาษให้ตรง (ตัด perspective) ด้วย ArUco — เขียน Python ใน Script TOP

Video Device In → Script TOP (ArUco + warp) → out · ต้องลง opencv-contrib-python

import cv2, numpy as np

def onCook(scriptOp):
    src  = scriptOp.inputs[0].numpyArray(delImageData=True)
    img  = np.flipud((src[:,:,:3]*255).astype(np.uint8))   # TD ภาพกลับหัว
    gray = cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)

    adict    = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)
    detector = cv2.aruco.ArucoDetector(adict, cv2.aruco.DetectorParameters())
    corners, ids, _ = detector.detectMarkers(gray)          # OpenCV >= 4.7

    if ids is not None and len(ids) >= 4:
        pts = order_by_id(corners, ids)                     # helper เขียนเอง
        W = H = 1024
        dst = np.float32([[0,0],[W,0],[W,H],[0,H]])
        M   = cv2.getPerspectiveTransform(pts, dst)
        warped = cv2.warpPerspective(img, M, (W,H))
        out = np.dstack([warped, np.full((H,W),255,np.uint8)]).astype(np.float32)/255
        scriptOp.copyNumpyArray(np.flipud(out))
    return

illustrative — ปรับตาม OpenCV version จริง

def onHTTPRequest(webServerDAT, request, response):
